## 최고의 집합
* 문제 링크 : https://school.programmers.co.kr/learn/courses/30/lessons/12938
* 시간 복자도 : O(n)
```java
import java.util.*;

class Solution {
    public int[] solution(int n, int s) {
        int size = n;
        int[] answer = new int[size];
        
        for(int i=size-1; i>=0; i--) {
            int res = s / n;
            if (s % n > 0) {
                res++;
            }
            answer[i] = res;
            s -= res;
            n--;
            if (s <= 0 && n > 0) {
                answer = new int[]{-1};
                break;
            }
        }
        
        return answer;
    }
}
```
