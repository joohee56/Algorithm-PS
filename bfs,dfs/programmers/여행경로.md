## 여행경로
* 문제 링크 : https://school.programmers.co.kr/learn/courses/30/lessons/43164
```java
import java.util.*;

class Solution {
    private String[][] tickets;
    private Map<String, PriorityQueue<String>> ticketMap;
    private String[] answer;
    
    public String[] solution(String[][] tickets) {
        this.tickets = tickets;
        this.ticketMap = new HashMap<>();
        
        for (String[] ticket : tickets) {
            PriorityQueue<String> pq = ticketMap.getOrDefault(ticket[0], new PriorityQueue<>());
            pq.offer(ticket[1]);
            ticketMap.put(ticket[0], pq);
        }
        
        answer = new String[tickets.length+1];
        List<String> path = new ArrayList<>();
        path.add("ICN");
        dfs(tickets.length+1, path, "ICN");
        return answer;
    }
    
    void dfs(int targetCnt, List<String> path, String now) {
        if (path.size() == targetCnt) {
            answer = path.toArray(new String[path.size()]);
            return;
        }
        
        PriorityQueue<String> pq = ticketMap.get(now);
        if (pq == null || pq.isEmpty()) return;
        String next = pq.poll();
        List<String> nPath = new ArrayList<>(path);
        nPath.add(next);
        dfs(targetCnt, nPath, next);
        pq.offer(next);
    }
}
```
